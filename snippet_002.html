SELECT @TotalPop = COUNT(DISTINCT PROCHI)
FROM Project_12345..Demography_Current

       COUNT(DISTINCT PROCHI) AS PatientCount,
       COUNT(DISTINCT PROCHI) * 100.0 / @TotalPop AS Pct
FROM Project_12345..Demography_Current
GROUP BY sex
